実行結果の空欄を埋めよ。

      console.log(typeof "a"); // > [①]
      console.log(typeof new String("a")); // > [②]
      console.log(typeof 1); // > [③]
      console.log(typeof new Number(1)); // > [④]

      console.log("a" instanceof String); // > [⑤]
      console.log("a" instanceof Object); // > [⑥]
      console.log(new String("a") instanceof String); // > [⑦]
      console.log(new String("a") instanceof Object); // > [⑧]

      console.log(1 instanceof Number); // > [⑨]
      console.log(1 instanceof Object); // > [⑩]
      console.log(new Number(1) instanceof Number); // > [(11)]
      console.log(new Number(1) instanceof Object); // > [(12)]
